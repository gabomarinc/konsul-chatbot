    );

    }
    
    console.log('âœ… Chat encontrado:', gabrielChat.name);

    // Obtener campos disponibles

    const fieldsResult = await api.getCustomFields();

    if (fieldsResult.success) {

    // Buscar valores en el chat
    console.log('\nğŸ” Buscando valores en el chat...');
    const buscarValores = (obj, path = '') => {
        const resultados = [];
        for (const key in obj) {
            const currentPath = path ? `${path}.${key}` : key;
            const value = obj[key];
            
            if (key.toLowerCase().includes('custom') || 
                key.toLowerCase().includes('field') ||
                key.toLowerCase().includes('perfil') ||
                key.toLowerCase().includes('dui') ||
                key.toLowerCase().includes('constancia')) {
                resultados.push({ path: currentPath, value: value });
            }
            
            if (value && typeof value === 'object' && !Array.isArray(value)) {
                resultados.push(...buscarValores(value, currentPath));
            }
        }
        return resultados;
    };
    
    const valoresEncontrados = buscarValores(gabrielChat);
